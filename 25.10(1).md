Определите очередной год, в который заданное число будет приходиться на пятницу.

```
#include <iostream>

int main()
{
    int day, month, a, y, d = 0, m;
    do {
        std::cout << "Input day: ";
        std::cin >> day;
    } while (day < 0 && day > 32);
    do {
        std::cout << "Input month: ";
        std::cin >> month;
    } while (month < 0 && month > 13);
    a = (14 - month) / 12;
    m = month + 12 * a - 2;
    for (int i = 2021; d != 5; --i) {
        y = i - a;
        d = (day + y + y / 4 - y / 100 + y / 400 + 31 * m / 12) % 7;
        if (d == 5) std::cout << "The desired year: " << i << "\n";
    }
}
```
