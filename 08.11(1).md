Заполнить массив nxm (размеры вводит пользователь) числами от 1 до n*m по спирали (улитке).

```
#include <iostream>


int main() {
	int n, m, a = 0;
	do {
		std::cout << "Input n: ";
		std::cin >> n;
	} while (n < 2);
	do {
		std::cout << "Input m: ";
		std::cin >> m;
	} while (m < 2);

	int** arr = new int* [n];
	for (int i = 0; i < n; ++i) {
		arr[i]  = new int[m];
	}
	int i = 0, j = 0;
	for (int nm = 0; a < n * m; ++nm) {
		for (; i < m - 1 - nm; ++i) arr[j][i] = ++a;
		for (; j < n - 1 - nm; ++j) arr[j][i] = ++a;
		for (; i > nm; --i) arr[j][i] = ++a;
		for (; j > nm; --j) arr[j][i] = ++a;
		i = nm + 1;
		j = nm + 1;
	}

	for (int x = 0; x < n; ++x) {
		for (int y = 0; y < m; ++y) {
			std::cout << arr[x][y] << "  ";
		}
		std::cout << "\n";
	}
}
```
