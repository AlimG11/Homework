#include <iostream>

int main()
{
    int month, a, y, d = 0, m, n = 0, p = 0;
    do {
        std::cout << "Input month: ";
        std::cin >> month;
    } while (month < 0 && month > 13);
    a = (14 - month) / 12;
    for (int i = 2000; i != 1900; --i) {
        y = i - a;
        for (int j = 1; j < 13; ++j) {
            m = j + 12 * a - 2;
            d = (13 + y + y / 4 - y / 100 + y / 400 + 31 * m / 12) % 7;
            n = d == 5 ? n + 1 : n;
            if (d == 5 && month == j) ++p;
        }
    }
    std::cout << "Number of Fridays: " << n << "\nProbability: " << (double)p / (double)n * 100. << "%\n";
}
