#include <iostream>

int main()
{
    int n, m = 1;
    do {
        std::cout << "Input number room: ";
        std::cin >> n;
    } while (n < 1);
    for (int i = 1; i <= n; ++i) {
        for (int j = 1; j <= i; ++j) {
            n -= i;
            ++m;
            if (n <= i) {
                std::cout << "Number floor: " << m << std::endl << "Number room: " << n << std::endl;
                return 0;
            }
        }
    }
    return 0;
}
